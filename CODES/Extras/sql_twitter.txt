(

    stemmed string,
    polarity string
)

CREATE EXTERNAL TABLE dictionary (
    type string,
    length int,
    word string,
    pos string,
    stemmed string,
    polarity string
)
ROW FORMAT DELIMITED FIELDS TERMINATED BY '	'
STORED AS TEXTFILE
LOCATION '/user/maria_dev/dictionary';

load data inpath '/user/maria_dev/dictionary.tsv' into TABLE dictionary;

create view temp_3 as select
id,
temp_2.tweet,
temp_2.word,
case s_d.polarity
            when  'negative' then -1
            when 'positive' then 1
            else 0 
end as polarity
from temp_2 left outer join dictionary s_d on temp_2.word = s_d.word;

-- Creating a summary view to check the total positive and negative words in each tweet
create view tweet_sentiment_summary as
select
id,
tweet,
sum(case when polarity = 1 then 1 else 0 end) as positive_words,
sum(case when polarity = -1 then 1 else 0 end) as negative_words
from temp_3
group by id, tweet;

-- Checking the sentiment summary result
SELECT * FROM tweet_sentiment_summary limit 25;
